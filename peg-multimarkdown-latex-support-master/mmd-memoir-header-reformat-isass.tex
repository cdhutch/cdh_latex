% Formatting for blue text box
\newcommand\bluebox[1]{\fcolorbox[cmyk]{0,0,0,0.5}{0.3659,0.1351,0.0077,0}{#1}}

% Formatting for capability two-pager headers
\newcommand\capabilityheader[2]{
\begin{minipage}[t]{\linewidth}
  \bluebox{
    \begin{minipage}{3.25in}
      \paragraph{Description}
      \label{description}
      {#2}
    \end{minipage}
    }
  \begin{minipage}{3.25in}
    \begin{figure}[H]
      \includegraphics[keepaspectratio,width=2.5in,height=0.75\textheight]{#1}
      \label{#1}
    \end{figure}
  \end{minipage}
\end{minipage}
}
